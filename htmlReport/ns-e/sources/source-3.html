 import jakarta.persistence.*;

 import java.time.OffsetDateTime;
 
 @Entity(name = "orders")
 @NoArgsConstructor

 @Getter
 @Setter
 @EqualsAndHashCode(onlyExplicitlyIncluded = true)
 @ToString
 public class OrderEntity {

     private String orderId;
